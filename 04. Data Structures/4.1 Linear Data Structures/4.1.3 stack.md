# 3. 스택 (Stack)

- **LIFO 원리** : Last In First Out, 가장 나중에 들어간 데이터가 가장 먼저 나옴
- 재귀적인 함수, 알고리즘에 사용되며 웹 브라우저 방문 기록 등에 쓰임.

| 시간복잡도    |   접근    |   탐색     |   삽입     |   삭제    |
| ----------- | -----=--- | --------- | --------- | --------- |
| **스택**   |    O(n)    |    O(n)    |   O(1)   |   O(1)    |

### 스택 구현

<details>
<summary> java 배열 기반 </summary>
<div markdown="1">

```java
class ArrayStack {
    private int[] stack;
    private int top;
    private int maxSize;

    public ArrayStack(int size) {
        maxSize = size;
        stack = new int[maxSize];
        top = -1;
    }

    public void push(int data) {
        if (top < maxSize - 1) {
            stack[++top] = data;
        } else {
            throw new RuntimeException("Stack Overflow");
        }
    }

    public int pop() {
        if (top >= 0) {
            return stack[top--];
        } else {
            throw new RuntimeException("Stack Underflow");
        }
    }

    public int peek() {
        if (top >= 0) {
            return stack[top];
        } else {
            throw new RuntimeException("Stack is Empty");
        }
    }

    public boolean isEmpty() {
        return top == -1;
    }

    public boolean isFull() {
        return top == maxSize - 1;
    }

    public int size() {
        return top + 1;
    }
}
```

```java
class LinkedStack {
    private ListNode top;
    private int size;

    public LinkedStack() {
        this.top = null;
        this.size = 0;
    }

    public void push(int data) {
        ListNode newNode = new ListNode(data);
        newNode.next = top;
        top = newNode;
        size++;
    }

    public int pop() {
        if (top == null) {
            throw new RuntimeException("Stack Underflow");
        }
        int data = top.data;
        top = top.next;
        size--;
        return data;
    }

    public int peek() {
        if (top == null) {
            throw new RuntimeException("Stack is Empty");
        }
        return top.data;
    }

    public boolean isEmpty() {
        return top == null;
    }

    public int size() {
        return size;
    }
}
```

```cpp
#include <iostream>

class Stack {
private:
    int* arr;       // 스택을 저장할 배열 포인터
    int capacity;   // 스택의 최대 크기
    int topIndex;   // 현재 top의 인덱스

public:
    // 생성자: 스택의 크기를 받아 배열을 생성하고 초기화
    Stack(int size) {
        capacity = size;           // 최대 크기를 설정
        arr = new int[capacity];   // 동적 배열 생성
        topIndex = -1;             // 스택이 비어있음을 나타냄
    }

    // 소멸자: 동적 할당된 배열 해제
    ~Stack() {
        delete[] arr;
    }

    // 스택이 비었는지 확인
    bool isEmpty() {
        return topIndex == -1;
    }

    // 스택이 가득 찼는지 확인
    bool isFull() {
        return topIndex == capacity - 1;
    }

    // 스택에 값 삽입 (push)
    void push(int val) {
        if (isFull()) {
            std::cout << "스택이 가득 찼습니다." << std::endl;
            return;
        }
        topIndex++;         // topIndex를 1 증가
        arr[topIndex] = val; // top 위치에 값 저장
    }

    // 스택에서 top 값 삭제 (pop)
    void pop() {
        if (isEmpty()) {
            std::cout << "스택이 비어있습니다." << std::endl;
            return;
        }
        topIndex--; // topIndex를 1 감소시켜 삭제 효과
    }

    // 스택의 top 값 조회 (peek)
    int top() {
        if (isEmpty()) {
            throw std::runtime_error("스택이 비어있습니다.");
        }
        return arr[topIndex]; // 현재 topIndex 위치의 값 반환
    }

    // 스택의 모든 요소 출력
    void printStack() {
        if (isEmpty()) {
            std::cout << "스택이 비어있습니다." << std::endl;
            return;
        }
        std::cout << "스택 내용 (bottom -> top): ";
        for (int i = 0; i <= topIndex; i++) {
            std::cout << arr[i] << " ";
        }
        std::cout << std::endl;
    }
};

// main 함수: 사용 예시
int main() {
    Stack stack(5); // 크기 5인 스택 생성

    stack.push(10); // 10 삽입
    stack.push(20); // 20 삽입
    stack.push(30); // 30 삽입

    stack.printStack(); // 출력: 10 20 30

    std::cout << "Top: " << stack.top() << std::endl; // 30

    stack.pop(); // top 삭제 (30 삭제)
    stack.printStack(); // 출력: 10 20

    stack.pop(); // 20 삭제
    stack.pop(); // 10 삭제
    stack.pop(); // 스택이 비어있습니다.

    return 0;
}
```

```cpp
#include <iostream>

// Node 구조체 정의: 값(data)과 다음 노드를 가리키는 포인터(next)
struct Node {
    int data;     // 노드가 저장하는 값
    Node* next;   // 다음 노드를 가리키는 포인터

    // 생성자: data를 초기화하고 next는 nullptr로 설정
    Node(int val) : data(val), next(nullptr) {}
};

// Stack 클래스 정의
class Stack {
private:
    Node* topNode; // 스택의 top을 가리키는 포인터

public:
    // 생성자: topNode를 nullptr로 초기화 (빈 스택)
    Stack() : topNode(nullptr) {}

    // 소멸자: 동적 할당된 모든 노드를 해제
    ~Stack() {
        while (topNode != nullptr) {
            Node* temp = topNode; // 현재 topNode를 temp에 저장
            topNode = topNode->next; // topNode를 다음 노드로 이동
            delete temp; // 기존 topNode 삭제
        }
    }

    // 스택이 비었는지 확인
    bool isEmpty() {
        return topNode == nullptr;
    }

    // 스택에 값 삽입 (push)
    void push(int val) {
        Node* newNode = new Node(val); // 새 노드 생성
        newNode->next = topNode; // 새 노드의 next를 현재 topNode로 연결
        topNode = newNode; // topNode를 새 노드로 변경
    }

    // 스택에서 top 값 삭제 (pop)
    void pop() {
        if (isEmpty()) {
            std::cout << "스택이 비어있습니다." << std::endl;
            return;
        }
        Node* temp = topNode; // 현재 topNode를 temp에 저장
        topNode = topNode->next; // topNode를 다음 노드로 이동
        delete temp; // 기존 topNode 삭제
    }

    // 스택의 top 값 조회 (peek)
    int top() {
        if (isEmpty()) {
            throw std::runtime_error("스택이 비어있습니다.");
        }
        return topNode->data; // 현재 topNode의 data 반환
    }

    // 스택의 모든 요소 출력
    void printStack() {
        if (isEmpty()) {
            std::cout << "스택이 비어있습니다." << std::endl;
            return;
        }
        std::cout << "스택 내용 (top -> bottom): ";
        Node* current = topNode; // topNode부터 시작
        while (current != nullptr) {
            std::cout << current->data << " "; // 현재 노드 출력
            current = current->next; // 다음 노드로 이동
        }
        std::cout << std::endl;
    }
};

// main 함수: 사용 예시
int main() {
    Stack stack; // 스택 생성

    stack.push(10); // 10 삽입
    stack.push(20); // 20 삽입
    stack.push(30); // 30 삽입

    stack.printStack(); // 출력: 30 20 10

    std::cout << "Top: " << stack.top() << std::endl; // 30

    stack.pop(); // top 삭제 (30 삭제)
    stack.printStack(); // 출력: 20 10

    stack.pop(); // 20 삭제
    stack.pop(); // 10 삭제
    stack.pop(); // 스택이 비어있습니다.

    return 0;
}
```

```java
public boolean isValidParentheses(String s) {
    Stack<Character> stack = new Stack<>();
    Map<Character, Character> pairs = new HashMap<>();
    pairs.put('(', ')');
    pairs.put('[', ']');
    pairs.put('{', '}');

    for (char c : s.toCharArray()) {
        if (pairs.containsKey(c)) {
            stack.push(c);
        } else if (pairs.containsValue(c)) {
            if (stack.isEmpty() || pairs.get(stack.pop()) != c) {
                return false;
            }
        }
    }

    return stack.isEmpty();
}
```

**3. 후위 표기법 계산**

```java
public int evaluatePostfix(String expression) {
    Stack<Integer> stack = new Stack<>();
    String[] tokens = expression.split(" ");

    for (String token : tokens) {
        if (isOperator(token)) {
            int b = stack.pop();
            int a = stack.pop();
            stack.push(calculate(a, b, token));
        } else {
            stack.push(Integer.parseInt(token));
        }
    }

    return stack.pop();
}

private boolean isOperator(String token) {
    return token.equals("+") || token.equals("-") ||
           token.equals("*") || token.equals("/");
}

private int calculate(int a, int b, String operator) {
    switch (operator) {
        case "+": return a + b;
        case "-": return a - b;
        case "*": return a * b;
        case "/": return a / b;
        default: throw new IllegalArgumentException("Invalid operator");
    }
}
```
